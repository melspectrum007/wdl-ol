220  IPattern GetSVGPattern(const NSVGpaint& paint, float opacity)
221  {
222  int alpha = std::min(255, std::max(int(roundf(opacity * 255.f)), 0));
223 
224  switch (paint.type)
225  {
226  case NSVG_PAINT_COLOR:
227  return IColor(alpha, (paint.color >> 0) & 0xFF, (paint.color >> 8) & 0xFF, (paint.color >> 16) & 0xFF);
228 
229  case NSVG_PAINT_LINEAR_GRADIENT:
230  case NSVG_PAINT_RADIAL_GRADIENT:
231  {
232  NSVGgradient* pGrad = paint.gradient;
233 
234  IPattern pattern(paint.type == NSVG_PAINT_LINEAR_GRADIENT ? kLinearPattern : kRadialPattern);
235 
236  // Set Extend Rule
237  switch (pGrad->spread)
238  {
239  case NSVG_SPREAD_PAD: pattern.mExtend = kExtendPad; break;
240  case NSVG_SPREAD_REFLECT: pattern.mExtend = kExtendReflect; break;
241  case NSVG_SPREAD_REPEAT: pattern.mExtend = kExtendRepeat; break;
242  }
243 
244  // Copy Stops
245  for (int i = 0; i < pGrad->nstops; i++)
246  {
247  int color = pGrad->stops[i].color;
248  pattern.AddStop(IColor(255, (color >> 0) & 0xFF, (color >> 8) & 0xFF, (color >> 16) & 0xFF), pGrad->stops[i].offset);
249  }
250 
251  // Copy transform
252  pattern.SetTransform(pGrad->xform[0], pGrad->xform[1], pGrad->xform[2], pGrad->xform[3], pGrad->xform[4], pGrad->xform[5]);
253 
254  return pattern;
255  }
256  default:
257  return IColor(alpha, 0, 0, 0);
258  }
259  }

261  void RenderNanoSVG(NSVGimage* pImage)
262  {
263  for (NSVGshape* pShape = pImage->shapes; pShape; pShape = pShape->next)
264  {
265  if (!(pShape->flags & NSVG_FLAGS_VISIBLE))
266  continue;
267 
268  for (NSVGpath* pPath = pShape->paths; pPath; pPath = pPath->next)
269  {
270  PathMoveTo(pPath->pts[0], pPath->pts[1]);
271 
272  for (int i = 0; i < pPath->npts - 1; i += 3)
273  {
274  float *p = pPath->pts + i * 2 + 2;
275  PathCurveTo(p[0], p[1], p[2], p[3], p[4], p[5]);
276  }
277 
278  if (pPath->closed)
279  PathClose();
280  }
281 
282  // Fill
283  if (pShape->fill.type != NSVG_PAINT_NONE)
284  {
285  IFillOptions options;
286 
287  if (pShape->fillRule == NSVG_FILLRULE_EVENODD)
288  options.mFillRule = kFillEvenOdd;
289  else
290  options.mFillRule = kFillWinding;
291 
292  options.mPreserve = pShape->stroke.type != NSVG_PAINT_NONE;
293  PathFill(GetSVGPattern(pShape->fill, pShape->opacity), options, nullptr);
294  }
295 
296  // Stroke
297  if (pShape->stroke.type != NSVG_PAINT_NONE)
298  {
299  IStrokeOptions options;
300 
301  options.mMiterLimit = pShape->miterLimit;
302 
303  switch (pShape->strokeLineCap)
304  {
305  case NSVG_CAP_BUTT: options.mCapOption = kCapButt; break;
306  case NSVG_CAP_ROUND: options.mCapOption = kCapRound; break;
307  case NSVG_CAP_SQUARE: options.mCapOption = kCapSquare; break;
308  }
309 
310  switch (pShape->strokeLineJoin)
311  {
312  case NSVG_JOIN_MITER: options.mJoinOption = kJoinMiter; break;
313  case NSVG_JOIN_ROUND: options.mJoinOption = kJoinRound; break;
314  case NSVG_JOIN_BEVEL: options.mJoinOption = kJoinBevel; break;
315  }
316 
317  options.mDash.SetDash(pShape->strokeDashArray, pShape->strokeDashOffset, pShape->strokeDashCount);
318 
319  PathStroke(GetSVGPattern(pShape->stroke, pShape->opacity), pShape->strokeWidth, options, nullptr);
320  }
321  }
